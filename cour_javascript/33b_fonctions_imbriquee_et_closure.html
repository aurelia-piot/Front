Les fonctions imbriquées et closures

ont peux appeler une fonctions depuis n'importe où meme si elle se trouve a l'interieure d'une autre fonctions

les fonctions imbriquéessont souvent utiliser comme uin moyen de garder du code bien organisé. Une grande fonction complex est facile à comprendre si on la divise en plusieurs sous-fonctions. 


//Etape 2
//on va créer une meta fonction qui imbrique plusieurs sous-fonctions
    function maMetaFonction(){
        var maVariable = 1;

        sousFonction1(maVariable);
        sousFonction2(maVariable);
        sousFonction3(maVariable);
    }//fin de maMetaFonction

    function sousFonction1(valeur){
        document.write(valeur);
    }
    function sousFonction2(valeur) {
        document.write(valeur);
    }
    function sousFonction3(valeur) {
        document.write(valeur);
    }

//ici on declanche maMetaFunction

maMetaFonction();

//ici nous avons bien accès à la variable maVariable car ont a passer sa valeur dans la metaFunction et les
sous-fonctions en tant que parametre.
